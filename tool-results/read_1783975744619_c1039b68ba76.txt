  1407→export async function seedPhones(): Promise<{
  1408→  brands: number;
  1409→  phones: number;
  1410→  specs: number;
  1411→  benchmarks: number;
  1412→  prices: number;
  1413→  skipped: number;
  1414→}> {
  1415→  await connectDB();
  1416→
  1417→  let brands = 0;
  1418→  let phones = 0;
  1419→  let specs = 0;
  1420→  let benchmarks = 0;
  1421→  let prices = 0;
  1422→  let skipped = 0;
  1423→
  1424→  /* ---- 1. Upsert brands ---- */
  1425→  const brandMap = new Map<string, Types.ObjectId>();
  1426→
  1427→  for (const b of BRANDS) {
  1428→    const doc = await (Brand as any).findOneAndUpdate(
  1429→      { slug: b.slug },
  1430→      {
  1431→        name: b.name,
  1432→        slug: b.slug,
  1433→        logo: '',
  1434→        country: b.country,
  1435→        description: b.description,
  1436→        sortOrder: b.sortOrder,
  1437→        active: true,
  1438→      },
  1439→      { upsert: true, new: true, setDefaultsOnInsert: true },
  1440→    );
  1441→    brandMap.set(b.name, doc._id as Types.ObjectId);
  1442→    brands++;
  1443→  }
  1444→
  1445→  /* ---- 2. Upsert phones + related docs ---- */
  1446→  for (const p of PHONES) {
  1447→    const brandId = brandMap.get(p.brand);
  1448→    if (!brandId) {
  1449→      console.warn(`[seed] Skipping "${p.modelName}" — brand "${p.brand}" not found`);
  1450→      skipped++;
  1451→      continue;
  1452→    }
  1453→
  1454→    // Upsert phone
  1455→    const phoneDoc = await (Phone as any).findOneAndUpdate(
  1456→      { slug: p.slug },
  1457→      {
  1458→        brandId,
  1459→        modelName: p.modelName,
  1460→        slug: p.slug,
  1461→        releaseDate: p.releaseDate,
  1462→        pricePKR: p.pricePKR,
  1463→        ptaStatus: p.ptaStatus,
  1464→        ptaApproved: p.ptaStatus === 'PTA Approved',
  1465→        featured: p.featured,
  1466→        trending: p.trending,
  1467→        upcoming: p.upcoming,
  1468→        thumbnail: '',
  1469→        description: p.description,
  1470→        cameraScore: p.cameraScore,
  1471→        performanceScore: p.performanceScore,
  1472→        batteryScore: p.batteryScore,
  1473→        displayScore: p.displayScore,
  1474→        valueScore: p.valueScore,
  1475→        overallRating: p.overallRating,
  1476→        active: true,
  1477→        status: 'published',
  1478→        pros: p.pros,
  1479→        cons: p.cons,
  1480→        reviewSummary: p.reviewSummary,
  1481→        reviewVerdict: p.reviewVerdict,
  1482→      },
  1483→      { upsert: true, new: true, setDefaultsOnInsert: true },
  1484→    );
  1485→    phones++;
  1486→
  1487→    const phoneId = phoneDoc._id as Types.ObjectId;
  1488→
  1489→    // Upsert specs
  1490→    const specsData: Record<string, string> = {};
  1491→    for (const [key, val] of Object.entries(p.specs)) {
  1492→      specsData[key] = (val as string) ?? '';
  1493→    }
  1494→    await (PhoneSpecs as any).findOneAndUpdate(
  1495→      { phoneId },
  1496→      { phoneId, ...specsData },
  1497→      { upsert: true, new: true, setDefaultsOnInsert: true },
  1498→    );
  1499→    specs++;
  1500→
  1501→    // Upsert benchmark
  1502→    if (p.benchmarks) {
  1503→      await (PhoneBenchmark as any).findOneAndUpdate(
  1504→        { phoneId },
  1505→        {
  1506→          phoneId,
  1507→          antutu: p.benchmarks.antutu ?? 0,
  1508→          geekbenchSingle: p.benchmarks.geekbenchSingle ?? 0,
  1509→          geekbenchMulti: p.benchmarks.geekbenchMulti ?? 0,
  1510→          gamingScore: p.benchmarks.gamingScore ?? 0,
  1511→          pubgFps: p.benchmarks.pubgFps ?? '',
  1512→          codMobileFps: p.benchmarks.codMobileFps ?? '',
  1513→          genshinFps: p.benchmarks.genshinFps ?? '',
  1514→          videoPlayback: p.benchmarks.videoPlayback ?? '',
  1515→          gamingBattery: p.benchmarks.gamingBattery ?? '',
  1516→          browsingBattery: p.benchmarks.browsingBattery ?? '',
  1517→        },
  1518→        { upsert: true, new: true, setDefaultsOnInsert: true },
  1519→      );
  1520→      benchmarks++;
  1521→    }
  1522→
  1523→    // Handle prices: delete old, insert new
  1524→    if (p.prices && p.prices.length > 0) {
  1525→      await PhonePrice.deleteMany({ phoneId });
  1526→      const priceDocs = p.prices.map((pr) => ({
  1527→        phoneId,
  1528→        storeName: pr.storeName,
  1529→        price: pr.price,
  1530→        url: '',
  1531→        inStock: pr.inStock ?? true,
  1532→      }));
  1533→      const inserted = await PhonePrice.insertMany(priceDocs as any);
  1534→      prices += inserted.length;
  1535→    }
  1536→  }
  1537→
  1538→  const summary = { brands, phones, specs, benchmarks, prices, skipped };
  1539→  console.log('[seedPhones] Done:', JSON.stringify(summary, null, 2));
  1540→  return summary;
  1541→}
