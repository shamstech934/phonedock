    42тЖТ    42тЖТfunction useHashRouter() {
    43тЖТ    43тЖТ  const [view, setView] = useState<View>('home');
    44тЖТ    44тЖТ  const [params, setParams] = useState<Record<string, string>>({});
    45тЖТ    45тЖТ
    46тЖТ    46тЖТ  useEffect(() => {
    47тЖТ    47тЖТ    const parseHash = () => {
    48тЖТ    48тЖТ      const hash = window.location.hash.slice(1) || '/';
    49тЖТ    49тЖТ      const parts = hash.split('/').filter(Boolean);
    50тЖТ    50тЖТ      if (parts.length === 0) { setView('home'); setParams({}); return; }
    51тЖТ    51тЖТ      if (parts[0] === 'phone' && parts[1]) { setView('phone'); setParams({ slug: parts[1] }); return; }
    52тЖТ    52тЖТ      if (parts[0] === 'compare') { setView('compare'); setParams(Object.fromEntries(new URLSearchParams(parts.slice(1).join('/')))); return; }
    53тЖТ    53тЖТ      if (parts[0] === 'brand' && parts[1]) { setView('brand'); setParams({ slug: parts[1] }); return; }
    54тЖТ    54тЖТ      if (parts[0] === 'search') { setView('search'); setParams({ q: parts.slice(1).join('/') }); return; }
    55тЖТ    55тЖТ      if (parts[0] === 'brands') { setView('brands'); setParams({}); return; }
    56тЖТ    56тЖТ      if (parts[0] === 'news') { setView('news'); setParams({}); return; }
    57тЖТ    57тЖТ      if (parts[0] === 'admin' && parts[1] === 'login') { setView('admin-login'); setParams({}); return; }
    58тЖТ    58тЖТ      if (parts[0] === 'admin' && parts[1] === 'phones') { setView('admin-phones'); setParams({}); return; }
    59тЖТ    59тЖТ      if (parts[0] === 'admin' && parts[1] === 'brands') { setView('admin-brands'); setParams({}); return; }
    60тЖТ    60тЖТ      if (parts[0] === 'admin' && parts[1] === 'news') { setView('admin-news'); setParams({}); return; }
    61тЖТ    61тЖТ      if (parts[0] === 'admin' && parts[1] === 'sponsors') { setView('admin-sponsors'); setParams({}); return; }
    62тЖТ    62тЖТ      if (parts[0] === 'admin' && parts[1] === 'activity') { setView('admin-activity'); setParams({}); return; }
    63тЖТ    63тЖТ      if (parts[0] === 'admin' && parts[1] === 'dashboard') { setView('admin-dashboard'); setParams({}); return; }
    64тЖТ    64тЖТ      if (parts[0] === 'admin') { setView('admin'); setParams({}); return; }
    65тЖТ    65тЖТ      setView('home'); setParams({});
    66тЖТ    66тЖТ    };
    67тЖТ    67тЖТ    parseHash();
    68тЖТ    68тЖТ    window.addEventListener('hashchange', parseHash);
    69тЖТ    69тЖТ    return () => window.removeEventListener('hashchange', parseHash);
    70тЖТ    70тЖТ  }, []);
    71тЖТ    71тЖТ
    72тЖТ    72тЖТ  const navigate = useCallback((path: string) => {
    73тЖТ    73тЖТ    window.location.hash = path;
    74тЖТ    74тЖТ  }, []);
    75тЖТ    75тЖТ
    76тЖТ    76тЖТ  return { view, params, navigate };
    77тЖТ    77тЖТ}
