    43→function useHashRouter() {
    44→  const [view, setView] = useState<View>('home');
    45→  const [params, setParams] = useState<Record<string, string>>({});
    46→
    47→  useEffect(() => {
    48→    const parseHash = () => {
    49→      const hash = window.location.hash.slice(1) || '/';
    50→      const parts = hash.split('/').filter(Boolean);
    51→      if (parts.length === 0) { setView('home'); setParams({}); return; }
    52→      if (parts[0] === 'phone' && parts[1]) { setView('phone'); setParams({ slug: parts[1] }); return; }
    53→      if (parts[0] === 'compare') { setView('compare'); setParams(Object.fromEntries(new URLSearchParams(parts.slice(1).join('/')))); return; }
    54→      if (parts[0] === 'brand' && parts[1]) { setView('brand'); setParams({ slug: parts[1] }); return; }
    55→      if (parts[0] === 'search') { setView('search'); setParams({ q: parts.slice(1).join('/') }); return; }
    56→      if (parts[0] === 'brands') { setView('brands'); setParams({}); return; }
    57→      if (parts[0] === 'news') { setView('news'); setParams({}); return; }
    58→      if (parts[0] === 'admin' && parts[1] === 'login') { setView('admin-login'); setParams({}); return; }
    59→      if (parts[0] === 'admin' && parts[1] === 'phones') { setView('admin-phones'); setParams({}); return; }
    60→      if (parts[0] === 'admin' && parts[1] === 'brands') { setView('admin-brands'); setParams({}); return; }
    61→      if (parts[0] === 'admin' && parts[1] === 'news') { setView('admin-news'); setParams({}); return; }
    62→      if (parts[0] === 'admin' && parts[1] === 'sponsors') { setView('admin-sponsors'); setParams({}); return; }
    63→      if (parts[0] === 'admin' && parts[1] === 'activity') { setView('admin-activity'); setParams({}); return; }
    64→      if (parts[0] === 'admin' && parts[1] === 'dashboard') { setView('admin-dashboard'); setParams({}); return; }
    65→      if (parts[0] === 'admin') { setView('admin'); setParams({}); return; }
    66→      setView('home'); setParams({});
    67→    };
    68→    parseHash();
    69→    window.addEventListener('hashchange', parseHash);
    70→    return () => window.removeEventListener('hashchange', parseHash);
    71→  }, []);
    72→
    73→  const navigate = useCallback((path: string) => {
    74→    window.location.hash = path;
    75→  }, []);
    76→
    77→  return { view, params, navigate };
    78→}
