    42→function useHashRouter() {
    43→  const [view, setView] = useState<View>('home');
    44→  const [params, setParams] = useState<Record<string, string>>({});
    45→
    46→  useEffect(() => {
    47→    const parseHash = () => {
    48→      const hash = window.location.hash.slice(1) || '/';
    49→      const parts = hash.split('/').filter(Boolean);
    50→      if (parts.length === 0) { setView('home'); setParams({}); return; }
    51→      if (parts[0] === 'phone' && parts[1]) { setView('phone'); setParams({ slug: parts[1] }); return; }
    52→      if (parts[0] === 'compare') { setView('compare'); setParams(Object.fromEntries(new URLSearchParams(parts.slice(1).join('/')))); return; }
    53→      if (parts[0] === 'brand' && parts[1]) { setView('brand'); setParams({ slug: parts[1] }); return; }
    54→      if (parts[0] === 'search') { setView('search'); setParams({ q: parts.slice(1).join('/') }); return; }
    55→      if (parts[0] === 'brands') { setView('brands'); setParams({}); return; }
    56→      if (parts[0] === 'news') { setView('news'); setParams({}); return; }
    57→      if (parts[0] === 'admin' && parts[1] === 'login') { setView('admin-login'); setParams({}); return; }
    58→      if (parts[0] === 'admin' && parts[1] === 'phones') { setView('admin-phones'); setParams({}); return; }
    59→      if (parts[0] === 'admin' && parts[1] === 'brands') { setView('admin-brands'); setParams({}); return; }
    60→      if (parts[0] === 'admin' && parts[1] === 'news') { setView('admin-news'); setParams({}); return; }
    61→      if (parts[0] === 'admin' && parts[1] === 'sponsors') { setView('admin-sponsors'); setParams({}); return; }
    62→      if (parts[0] === 'admin' && parts[1] === 'activity') { setView('admin-activity'); setParams({}); return; }
    63→      if (parts[0] === 'admin' && parts[1] === 'dashboard') { setView('admin-dashboard'); setParams({}); return; }
    64→      if (parts[0] === 'admin') { setView('admin'); setParams({}); return; }
    65→      setView('home'); setParams({});
    66→    };
    67→    parseHash();
    68→    window.addEventListener('hashchange', parseHash);
    69→    return () => window.removeEventListener('hashchange', parseHash);
    70→  }, []);
    71→
    72→  const navigate = useCallback((path: string) => {
    73→    window.location.hash = path;
    74→  }, []);
    75→
    76→  return { view, params, navigate };
    77→}
