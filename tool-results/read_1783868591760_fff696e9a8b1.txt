    40→function useHashRouter() {
    41→  const [view, setView] = useState<View>('home');
    42→  const [params, setParams] = useState<Record<string, string>>({});
    43→
    44→  useEffect(() => {
    45→    const parseHash = () => {
    46→      const hash = window.location.hash.slice(1) || '/';
    47→      const parts = hash.split('/').filter(Boolean);
    48→      if (parts.length === 0) { setView('home'); setParams({}); return; }
    49→      if (parts[0] === 'phone' && parts[1]) { setView('phone'); setParams({ slug: parts[1] }); return; }
    50→      if (parts[0] === 'compare') { setView('compare'); setParams(Object.fromEntries(new URLSearchParams(parts.slice(1).join('/')))); return; }
    51→      if (parts[0] === 'brand' && parts[1]) { setView('brand'); setParams({ slug: parts[1] }); return; }
    52→      if (parts[0] === 'search') { setView('search'); setParams({ q: parts.slice(1).join('/') }); return; }
    53→      if (parts[0] === 'brands') { setView('brands'); setParams({}); return; }
    54→      if (parts[0] === 'news') { setView('news'); setParams({}); return; }
    55→      if (parts[0] === 'admin' && parts[1] === 'login') { setView('admin-login'); setParams({}); return; }
    56→      if (parts[0] === 'admin' && parts[1] === 'phones') { setView('admin-phones'); setParams({}); return; }
    57→      if (parts[0] === 'admin' && parts[1] === 'brands') { setView('admin-brands'); setParams({}); return; }
    58→      if (parts[0] === 'admin' && parts[1] === 'news') { setView('admin-news'); setParams({}); return; }
    59→      if (parts[0] === 'admin' && parts[1] === 'dashboard') { setView('admin-dashboard'); setParams({}); return; }
    60→      if (parts[0] === 'admin') { setView('admin'); setParams({}); return; }
    61→      setView('home'); setParams({});
    62→    };
    63→    parseHash();
    64→    window.addEventListener('hashchange', parseHash);
    65→    return () => window.removeEventListener('hashchange', parseHash);
    66→  }, []);
    67→
    68→  const navigate = useCallback((path: string) => {
    69→    window.location.hash = path;
    70→  }, []);
    71→
    72→  return { view, params, navigate };
    73→}
