    42→function useHashRouter() {
    43→  const [view, setView] = useState<View>('home');
    44→  const [params, setParams] = useState<Record<string, string>>({});
    45→
    46→  useEffect(() => {
    47→    const parseHash = () => {
    48→      const hash = window.location.hash.slice(1) || '/';
    49→      const parts = hash.split('/').filter(Boolean);
    50→      if (parts.length === 0) { setView('home'); setParams({}); return; }
    51→      if (parts[0] === 'phone' && parts[1]) { setView('phone'); setParams({ slug: parts[1] }); return; }
    52→      if (parts[0] === 'compare') { setView('compare'); setParams(Object.fromEntries(new URLSearchParams(parts.slice(1).join('/')))); return; }
    53→      if (parts[0] === 'brand' && parts[1]) { setView('brand'); setParams({ slug: parts[1] }); return; }
    54→      if (parts[0] === 'search') { setView('search'); setParams({ q: parts.slice(1).join('/') }); return; }
    55→      if (parts[0] === 'brands') { setView('brands'); setParams({}); return; }
    56→      if (parts[0] === 'news') { setView('news'); setParams({}); return; }
    57→      if (parts[0] === 'admin' && parts[1] === 'login') { setView('admin-login'); setParams({}); return; }
    58→      if (parts[0] === 'admin' && parts[1] === 'phones') { setView('admin-phones'); setParams({}); return; }
    59→      if (parts[0] === 'admin' && parts[1] === 'brands') { setView('admin-brands'); setParams({}); return; }
    60→      if (parts[0] === 'admin' && parts[1] === 'news') { setView('admin-news'); setParams({}); return; }
    61→      if (parts[0] === 'admin' && parts[1] === 'sponsors') { setView('admin-sponsors'); setParams({}); return; }
    62→      if (parts[0] === 'admin' && parts[1] === 'activity') { setView('admin-activity'); setParams({}); return; }
    63→      if (parts[0] === 'admin' && parts[1] === 'dashboard') { setView('admin-dashboard'); setParams({}); return; }
    64→      if (parts[0] === 'admin') { setView('admin'); setParams({}); return; }
    65→      setView('home'); setParams({});
    66→    };
    67→    parseHash();
    68→    window.addEventListener('hashchange', parseHash);
    69→    return () => window.removeEventListener('hashchange', parseHash);
    70→  }, []);
    71→
    72→  const navigate = useCallback((path: string) => {
    73→    window.location.hash = path;
    74→  }, []);
    75→
    76→  return { view, params, navigate };
    77→}

  1916→// ============ MAIN APP ============
  1917→export default function PhoneDockApp() {
  1918→  const { view, params, navigate } = useHashRouter();
  1919→  const [homeData, setHomeData] = useState<HomeData | null>(null);
  1920→  const [homeLoading, setHomeLoading] = useState(true);
  1921→  const [admin, setAdmin] = useState<AdminUser | null>(() => {
  1922→    if (typeof window === 'undefined') return null;
  1923→    const saved = localStorage.getItem('phonedock_admin');
  1924→    if (saved) { try { const a = JSON.parse(saved); return a.admin; } catch { return null; } }
  1925→    return null;
  1926→  });
  1927→  const [adminToken, setAdminToken] = useState<string | null>(() => {
  1928→    if (typeof window === 'undefined') return null;
  1929→    const saved = localStorage.getItem('phonedock_admin');
  1930→    if (saved) { try { const a = JSON.parse(saved); return a.token; } catch { return null; } }
  1931→    return null;
  1932→  });
  1933→  const { theme, setTheme } = useTheme();
  1934→
  1935→  useEffect(() => {
  1936→    const savedTheme = localStorage.getItem('phonedock_theme');
  1937→    if (savedTheme) { queueMicrotask(() => setTheme(savedTheme)); }
  1938→  }, []);
  1939→
  1940→  useEffect(() => {
  1941→    if (view === 'home') {
  1942→      let cancelled = false;
  1943→      fetch('/api/home').then(r => r.json()).then(d => { if (!cancelled) { setHomeData(d); setHomeLoading(false); } }).catch(() => { if (!cancelled) setHomeLoading(false); });
  1944→      return () => { cancelled = true; };
  1945→    }
  1946→  }, [view]);
  1947→
  1948→  const handleLogin = (a: AdminUser, token: string) => {
  1949→    setAdmin(a); setAdminToken(token);
  1950→    localStorage.setItem('phonedock_admin', JSON.stringify({ admin: a, token }));
  1951→  };
  1952→
  1953→  const toggleTheme = () => {
  1954→    const newTheme = theme === 'dark' ? 'light' : 'dark';
  1955→    setTheme(newTheme);
  1956→    localStorage.setItem('phonedock_theme', newTheme);
  1957→  };
  1958→
  1959→  const handleSearch = (q: string) => { navigate(`/search/${encodeURIComponent(q)}`); };
  1960→  const nav = (path: string) => { if (path === '/' || path === '') window.location.hash = '/'; else if (!path.startsWith('#')) window.location.hash = path; };
  1961→
  1962→  const handleLogout = () => {
  1963→    setAdmin(null); setAdminToken(null);
  1964→    localStorage.removeItem('phonedock_admin');
  1965→    nav('/');
  1966→  };
  1967→
  1968→  const isAdminView = view.startsWith('admin') && view !== 'admin-login';
  1969→  const showAdminSidebar = isAdminView && admin;
  1970→
  1971→  return (
  1972→    <div className="min-h-screen flex flex-col bg-background text-foreground">
  1973→      <Header onNavigate={nav} onSearch={handleSearch} theme={theme || 'light'} toggleTheme={toggleTheme} admin={admin} onLogout={handleLogout} />
  1974→
  1975→      <div className="flex flex-1">
  1976→        {showAdminSidebar && <AdminSidebar admin={admin!} onNavigate={nav} onLogout={handleLogout} currentView={view} />}
  1977→        <main className="flex-1 min-w-0">
  1978→          {view === 'home' && <HomePage data={homeData} loading={homeLoading} onNavigate={nav} />}
  1979→          {view === 'phone' && <PhoneDetailPage key={params.slug} slug={params.slug || ''} onNavigate={nav} />}
  1980→          {view === 'compare' && <ComparePage onNavigate={nav} />}
  1981→          {view === 'brands' && <BrandsPage onNavigate={nav} />}
  1982→          {view === 'brand' && <BrandDetailPage key={params.slug} slug={params.slug || ''} onNavigate={nav} />}
  1983→          {view === 'search' && <SearchPage key={params.q} query={params.q || ''} onNavigate={nav} />}
  1984→          {view === 'news' && <NewsPage onNavigate={nav} />}
  1985→          {view === 'admin-login' && <AdminLoginPage onLogin={handleLogin} />}
  1986→          {view === 'admin-dashboard' && admin && <AdminDashboard admin={admin} onNavigate={nav} />}
  1987→          {view === 'admin-phones' && admin && <AdminPhones onNavigate={nav} />}
  1988→          {view === 'admin-brands' && admin && <AdminBrands onNavigate={nav} />}
  1989→          {view === 'admin-news' && admin && <AdminNews onNavigate={nav} />}
  1990→          {view === 'admin-sponsors' && admin && <AdminSponsors onNavigate={nav} />}
  1991→          {view === 'admin-activity' && admin && <AdminActivity onNavigate={nav} />}
  1992→          {view === 'admin' && !admin && <AdminLoginPage onLogin={handleLogin} />}
  1993→        </main>
  1994→      </div>
  1995→
  1996→      {!isAdminView && <Footer onNavigate={nav} />}
  1997→    </div>
  1998→  );
